using System;
using System.Linq;
using Microsoft.EntityFrameworkCore;

namespace UserAuthApp
{
    class Program
    {
        static void Main(string[] args)
        {
            using var db = new AppDbContext();
            db.Database.EnsureCreated();

            Console.WriteLine("Введення логіна:");
            string? username = Console.ReadLine();

            Console.WriteLine("Введення пароля:");
            string? password = Console.ReadLine();

            if (string.IsNullOrWhiteSpace(username) || string.IsNullOrWhiteSpace(password))
            {
                Console.WriteLine("Логін і пароль обов’язкові");
                return;
            }

            if (password.Length < 6)
            {
                Console.WriteLine("Пароль має містити мінімум 6 символів");
                return;
            }

            if (!password.Any(char.IsDigit))
            {
                Console.WriteLine("Пароль має містити хоча б одну цифру");
                return;
            }

            var user = db.Users.FirstOrDefault(u => u.Username == username);

            if (user == null)
            {
                db.Users.Add(new User
                {
                    Username = username,
                    Password = password,
                    CreatedAt = DateTime.Now
                });

                db.SaveChanges();
                Console.WriteLine("Користувача зареєстровано");
            }
            else
            {
                if (user.Password == password)
                {
                    Console.WriteLine("Успішний вхід");
                }
                else
                {
                    Console.WriteLine("Невірний пароль");
                }
            }
        }
    }

    public class User
    {
        public int Id { get; set; }
        public string Username { get; set; } = null!;
        public string Password { get; set; } = null!;
        public DateTime CreatedAt { get; set; }
    }

    public class AppDbContext : DbContext
    {
        public DbSet<User> Users => Set<User>();

        protected override void OnConfiguring(DbContextOptionsBuilder optionsBuilder)
        {
            optionsBuilder.UseSqlServer(
                "Server=localhost;Database=UserAuthDb;Trusted_Connection=True;TrustServerCertificate=True"
            );
        }
    }
}
